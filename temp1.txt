    <script>
        // --- CONFIGURATION ---
        // TODO: PASTE YOUR CLIENT ID HERE
        const CLIENT_ID = 'YOUR_GOOGLE_CLIENT_ID_HERE.apps.googleusercontent.com';
        const SCOPES = 'https://www.googleapis.com/auth/calendar https://www.googleapis.com/auth/userinfo.profile';

        let tokenClient;
        let accessToken = null;

        function initGIS() {
            tokenClient = google.accounts.oauth2.initTokenClient({
                client_id: CLIENT_ID,
                scope: SCOPES,
                callback: (tokenResponse) => {
                    accessToken = tokenResponse.access_token;
                    onLoginSuccess();
                },
            });
        }

        function handleAuth() { tokenClient.requestAccessToken(); }

        function onLoginSuccess() {
            document.getElementById('authorize_button').style.display = 'none';
            document.getElementById('content_section').style.display = 'flex';
            
            fetch('https://www.googleapis.com/oauth2/v1/userinfo?alt=json', {
                headers: { 'Authorization': `Bearer ${accessToken}` }
            }).then(res => res.json()).then(data => {
                document.getElementById('user_name').innerText = data.given_name;
                document.getElementById('user_image').src = data.picture;
                document.getElementById('greeting_text').innerText = `Welcome back, ${data.given_name}. What shall we do?`;
            });
        }

        // --- COMMAND LOGIC ---
        async function processCommand() {
            const input = document.getElementById('user_input').value.toLowerCase();
            const resultCard = document.getElementById('result-card');
            const resultText = document.getElementById('result-text');

            if (!accessToken) {
                alert("Please sign in first to use Hackathon PPA features.");
                return;
            }

            if (input.includes("meet") || input.includes("meeting")) {
                resultText.innerHTML = "Creating your secure meeting link...";
                resultCard.style.display = "block";
                await createGoogleMeet();
            } else {
                resultText.innerText = "Command not recognized. Try typing 'create meeting'.";
                resultCard.style.display = "block";
            }
        }

        async function createGoogleMeet() {
            const event = {
                'summary': 'Hackathon PPA Meeting',
                'description': 'Generated by Hackathon Personal Productivity Assistant',
                'start': { 'dateTime': new Date().toISOString() },
                'end': { 'dateTime': new Date(new Date().getTime() + 30*60000).toISOString() },
                'conferenceData': {
                    'createRequest': {
                        'requestId': "req" + Date.now(),
                        'conferenceSolutionKey': { 'type': 'hangoutsMeet' }
                    }
                }
            };

            try {
                const response = await fetch('https://www.googleapis.com/calendar/v3/calendars/primary/events?conferenceDataVersion=1', {
                    method: 'POST',
                    headers: {
                        'Authorization': `Bearer ${accessToken}`,
                        'Content-Type': 'application/json'
                    },
                    body: JSON.stringify(event)
                });

                const data = await response.json();

                if (data.conferenceData && data.conferenceData.entryPoints) {
                    const meetLink = data.conferenceData.entryPoints[0].uri;
                    document.getElementById('result-text').innerHTML = 
                        `<div style="color: #64748b; margin-bottom: 5px;">Success! Here is your room:</div>
                         <a href="${meetLink}" target="_blank" class="meet-link">
                            <svg width="20" height="20" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2"><path d="M15 10l5-5v14l-5-5"></path><rect x="2" y="6" width="13" height="12" rx="2"></rect></svg>
                            Join Google Meet
                         </a>`;
                } else {
                    document.getElementById('result-text').innerText = "Event created, but no Meet link generated.";
                }
            } catch (error) {
                console.error(error);
                document.getElementById('result-text').innerText = "Error accessing Calendar API.";
            }
        }

        window.onload = initGIS;
    </script>
